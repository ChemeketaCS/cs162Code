#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;



int main()
{

    vector<int> vec {12, 3, 42, 6, 6, 18, 6};

    for(int value : vec) {
        cout << value << " ";
    }
    cout <<endl;

    string greeting = "hello";

    for(char c : greeting) {
        cout << c << " ";
    }
    cout <<endl;


    int nums[] = {1, 2, 3, 4, 5};

    //Bring in each item as reference
    // so we can change it
    for(int& n : nums) {
        n += 1;
    }

    for(int n : nums) {
        cout << n << " ";
    }
    cout <<endl;


    return 0;
}
