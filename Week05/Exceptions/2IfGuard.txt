// Division by zero.

#include <iostream>

using namespace std;

int main()
{
    int dividend, divisor, quotient;

    cout << "Enter the dividend: ";
    cin >> dividend;
    cout << endl;

    cout << "Enter the divisor: ";
    cin >> divisor;
    cout << endl;

    if (divisor != 0)
    {
        quotient = dividend / divisor;
        cout << "Quotient = " << quotient
             << endl;
    }
    else
        cout << "Cannot divide by zero."
             << endl;

    return 0;
}
